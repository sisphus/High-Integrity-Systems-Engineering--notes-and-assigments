with Ada.Text_IO; with Ada.Integer_Text_IO;
procedure Main with SPARK_Mode is
type IntPtr is access Integer;
procedure Zero_First_Arg(X : in IntPtr; Y : in IntPtr)
with Pre => (X /= null and Y /= null),
Post => (Y.all = Y.all'Old and X.all = 0) is
begin
X.all := 0;
Ada.Text_IO.New_Line;
end Zero_First_Arg;
procedure Swap(X : in out IntPtr; Y : in out IntPtr)
with Post => (X = Y'Old and Y = X'Old)
is
Temp : IntPtr := X; -- Temp and X alias each other
begin
Ada.Integer_Text_IO.Put(Temp.all);
X := Y;
Y := Temp;
end Swap;
X : IntPtr := new Integer'(1);
Y : IntPtr := new Integer'(2);
-- Alias : IntPtr := X;
begin
Zero_First_Arg(X,Y);
Swap(X,Y);
pragma Assert (Y.all = 0 and X.all = 2);
if (Y.all /= 0 or X.all /= 2) then
Ada.Text_IO.Put_Line("Assertion violated");
else
Ada.Text_IO.Put_Line("Assertion holds!");
end if;
end Main;
